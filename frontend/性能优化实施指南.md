```typescript
import { memo } from 'react';
import { useNavigate } from 'react-router-dom';

interface WorkCardProps {
  work: Work;
  onExport?: (workId: string) => void;
  onDelete?: (workId: string) => void;
}

const WorkCard = memo(({ work, onExport, onDelete }: WorkCardProps) => {
  const navigate = useNavigate();
  
  return (
    <div 
      className="work-card"
      onClick={() => navigate(`/novel/editor?workId=${work.id}`)}
    >
      {/* ... */}
    </div>
  );
}, (prevProps, nextProps) => {
  // 自定义比较函数
  return (
    prevProps.work.id === nextProps.work.id &&
    prevProps.work.title === nextProps.work.title &&
    prevProps.work.updated_at === nextProps.work.updated_at
  );
});

WorkCard.displayName = 'WorkCard';
export default WorkCard;
```

### 6. 虚拟滚动（可选）

对于长列表，考虑使用虚拟滚动:

```typescript
import { useVirtualizer } from '@tanstack/react-virtual';

function VirtualizedList({ items }: { items: Work[] }) {
  const parentRef = useRef<HTMLDivElement>(null);
  
  const virtualizer = useVirtualizer({
    count: items.length,
    getScrollElement: () => parentRef.current,
    estimateSize: () => 200, // 估计每个项目的高度
  });
  
  return (
    <div ref={parentRef} style={{ height: '600px', overflow: 'auto' }}>
      <div style={{ height: `${virtualizer.getTotalSize()}px`, position: 'relative' }}>
        {virtualizer.getVirtualItems().map((virtualItem) => (
          <div
            key={virtualItem.key}
            style={{
              position: 'absolute',
              top: 0,
              left: 0,
              width: '100%',
              height: `${virtualItem.size}px`,
              transform: `translateY(${virtualItem.start}px)`,
            }}
          >
            <WorkCard work={items[virtualItem.index]} />
          </div>
        ))}
      </div>
    </div>
  );
}
```

## 性能监控

### 使用 React DevTools Profiler

1. 打开 Chrome DevTools
2. 切换到 "Profiler" 标签
3. 点击录制按钮
4. 执行需要分析的操作
5. 停止录制，查看性能分析

### 关键指标

- **渲染时间**: 每个组件的渲染时间
- **重渲染次数**: 组件不必要的重渲染
- **提交次数**: 状态更新导致的提交
